# Fix: Unread Count Bugs (3 issues)

## Root Causes

1. **Badge unreliable** — relies entirely on WS. If WS misses a message, badge is stale until reload. No fallback.
2. **Own message shows unread** — WS delivers own message back; if `sendMessage` is in-flight when WS arrives, conversation state is in flux. `unread_count` not explicitly reset for own sends.
3. **Read messages become unread** — `queueMarkRead` batches with 2s delay. If user deselects conversation before flush, or another WS message arrives, `unread_count` gets re-incremented from DB-stale state.

## Fixes

### Fix 1: Periodic conversation refresh (30s) with pending-read protection
**File**: `apps/web/src/stores/messengerStore.ts`

Add 30s interval. When applying server response, check `pendingRead` — if a conversation is queued for read, force `unread_count = 0` (server hasn't processed the read yet).

```typescript
let refreshTimer: ReturnType<typeof setInterval> | null = null;

function startRefreshTimer(): void {
  if (refreshTimer) return;
  refreshTimer = setInterval(() => {
    const { me, selectedConversationId } = useMessengerStore.getState();
    if (!me) return;
    messengerApi.listConversations().then((convs) => {
      const updatedConvs = convs.map(c => {
        if (pendingRead.has(c.id) || selectedConversationId === c.id) {
          return { ...c, unread_count: 0 };
        }
        return c;
      });
      useMessengerStore.setState({ conversations: updatedConvs });
    }).catch(() => {});
  }, 30000);
}
```

Call `startRefreshTimer()` in `init()`. Cleanup in `destroyMessenger()`.

### Fix 3: Allow retry of failed reads with max 3 attempts
**File**: `apps/web/src/stores/messengerStore.ts`

Add `flushRetries` map. On API failure, retry up to 3 times. After 3 failures, drop permanently.

```typescript
const flushRetries = new Map<string, number>(); // "read:convId" → attempts

function flushPending(): void {
    // delivered
    for (const [convId, msgId] of pendingDelivered) {
        if (lastFlushed.delivered.get(convId) !== msgId) {
            lastFlushed.delivered.set(convId, msgId);
            messengerApi.markDelivered(convId, msgId).catch(() => {
                lastFlushed.delivered.delete(convId);
            });
        }
    }
    pendingDelivered.clear();

    // read
    for (const [convId, msgId] of pendingRead) {
        if (lastFlushed.read.get(convId) !== msgId) {
            lastFlushed.read.set(convId, msgId);
            messengerApi.markRead(convId, msgId).catch(() => {
                const key = `read:${convId}`;
                const attempts = (flushRetries.get(key) ?? 0) + 1;
                flushRetries.set(key, attempts);
                if (attempts < 3) {
                    lastFlushed.read.delete(convId);
                    pendingRead.set(convId, msgId);
                } else {
                    pendingRead.delete(convId);
                    lastFlushed.read.delete(convId);
                    flushRetries.delete(key);
                }
            });
        }
    }
    pendingRead.clear();
}
```

Clean `flushRetries` in `destroyMessenger()`.

### Fix 4: Force unread=0 for selected conversation on WS message
**File**: `apps/web/src/stores/messengerStore.ts`

```typescript
updateConversationFromWs: (convId, updates, incrementUnread = false) => {
    const s = get();
    const found = s.conversations.some((c) => c.id === convId);
    if (!found) {
        s.ensureConversation(convId);
        return;
    }
    set((s2) => ({
        conversations: s2.conversations.map((c) => {
            if (c.id !== convId) return c;
            const updated = { ...c, ...updates };
            if (incrementUnread && s2.selectedConversationId !== convId) {
                updated.unread_count = (c.unread_count ?? 0) + 1;
            } else if (s2.selectedConversationId === convId) {
                updated.unread_count = 0;
            }
            return updated;
        }),
    }));
},
```
